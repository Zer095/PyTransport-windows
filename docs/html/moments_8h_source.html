126

133 if (i == j) // Diagonal elements

136 }
137 else
138 {
139 S[i + 2 * nF * j] = 0; // Off-diagonal field-field
140 }
141 }
142 else // Pi components
143 {
144 if (j == i)
145 {
146 S[i + 2 * nF * j] = (k * k) / (2. * k * k * k); // Initial condition for momentum-momentum correlation
147 }
148 else
149 {
150 S[i + 2 * nF * j] = 0; // Off-diagonal momentum-momentum
151 }
152 }
153 // Off-diagonal field-momentum and momentum-field components are initially zero
154 }
155 }

492 vector<double> M_matrix(2 * nF * 2 * nF); // M is a (2*nF) x (2*nF) matrix
493 double a = exp(N); // Scale factor
494 model m;
495 potential pot;
496 double ep = m.Ep(f, p); // Epsilon slow-roll parameter
497 double Hi = m.H(f, p); // Hubble rate
498 vector<double> dVVi = pot.dVV(f, p); // Second derivative of potential
499
500 for (int i = 0; i < nF; i++)

502 for (int j = 0; j < nF; j++)
503 {
504 // Fill elements of M
505 M_matrix[i + j * 2 * nF] = 0.;
506 M_matrix[i + (j + nF) * 2 * nF] = -1. / a / a; // Related to kinetic terms
507 M_matrix[i + nF + j * 2 * nF] = -a * a * dVVi[i + j * nF]; // Related to potential terms
508 M_matrix[i + nF + (j + nF) * 2 * nF] = -3. * Hi; // Damping term from Hubble expansion
509
510 if (i == j)
511 {
512 M_matrix[i + (j + nF) * 2 * nF] = M_matrix[i + (j + nF) * 2 * nF] - k * k / (Hi); // Wavenumber contribution
513 M_matrix[i + nF + (j + nF) * 2 * nF] = M_matrix[i + nF + (j + nF) * 2 * nF] - ep * Hi; // Epsilon contribution
514 }
515 }
