92 double H(vector<double> f, vector<double> p)
93 {
94 double Hi2=0;
95 double Vi;
96 Vi=pot.V(f,p);
97 for(int i=0; i<nF; i++)
98 {
99 Hi2 = Hi2 + 1./3.*(f[nF+i]*f[nF+i]/2.); // Contribution from kinetic energy
100 }
101 Hi2=Hi2 + 1./3.*Vi; // Contribution from potential energy
102 return sqrt(Hi2);
